#include<stdio.h>
#include<unistd.h>
#include<stdlib.h>
#include<string.h>


#define MAX_ALPHABET 26
#define MAX_LINES 500
#define MAX_LINE_LENGTH 200
#define ASCII_SA 97
#define ASCII_BA 65

char * palphbet [MAX_ALPHABET][MAX_LINES];
int group_count [MAX_ALPHABET];
char lines [MAX_LINES][MAX_LINE_LENGTH];

main () {
int i=0, j=0;
FILE *fp;
char first_char;

memset (palphbet, 0, MAX_ALPHABET*MAX_LINES*4);

fp = fopen("temp.txt", "r");
if (fp == NULL ) {
fprintf (stderr, "\ntemp.txt file not found\n");
exit (1);
}
while ( fgets(lines[i], MAX_LINE_LENGTH, fp) ) {
if (isalpha(lines[i][0])) i++;
if (i >= MAX_LINES) break;
}
fclose(fp);

i-=2;
while (i >= 0) {
first_char = tolower (lines[i][0]);
palphbet[first_char - ASCII_SA][group_count[first_char - ASCII_SA]++] = lines[i];
i--;
}

for (i=0; i<MAX_ALPHABET; i++) {
for (j=0; j<group_count[i]; j++) {
if (!j) printf ("\nNUMBER of LINES STARTING WITH CHARS \" %c OR %c\" = %d\n", ASCII_BA+i, ASCII_SA+i, group_count[i]);
printf ("%s", palphbet[i][j]);
}
}
}
