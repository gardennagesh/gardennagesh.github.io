#include<stdio.h>

#include<unistd.h>

main (int argc, char **argv) {

if (argc < 2) {

fprintf (stderr, "\nUsage: ./a.out cmd [options]...\n");

}

if (!fork()) {

argv++;

execvp(argv[0], argv);

}

}
